BUG #5: PROXY DISPLAY SHOWS FULL PROXY URL
================================================================================
Location: bot.go, lines 853-871
Severity: MEDIUM

BEFORE (Buggy):
    func maskProxy(proxyURL string) string {
        if proxyURL == "" {
            return "∅"
        }
        u, err := url.Parse(proxyURL)
        if err != nil {
            return proxyURL  // SECURITY ISSUE - returns full URL with credentials!
        }
        if u.User != nil {
            pass, _ := u.User.Password()
            if pass != "" {
                u.User = url.UserPassword(u.User.Username(), "******")
            }
        }
        return u.String()  // Still shows full URL structure
    }

AFTER (Fixed):
    func maskProxy(proxyURL string) string {
        if proxyURL == "" {
            return "∅"
        }
        u, err := url.Parse(proxyURL)
        if err != nil {
            // Try to extract just host:port from credentials
            if strings.Contains(proxyURL, "@") {
                parts := strings.Split(proxyURL, "@")
                if len(parts) > 1 {
                    return parts[len(parts)-1]  // Return only host:port
                }
            }
            return "***"  // Hide unparseable proxies
        }
        // Return only host:port, no credentials
        if u.Host != "" {
            return u.Host
        }
        return "***"
    }
